class Solution {
    public List<Integer> selfDividingNumbers(int left, int right) {
        List<Integer> res = new LinkedList<>();
        while(left <= right){
            if(isSelfDividingNumbers(left)){
                res.add(left);
            }
            left++;
        }
        return res;
    }
    
    private boolean isSelfDividingNumbers(int num){
    	int digit = 0;
    	int n = num;
    	while(num > 0){
    		digit = num % 10;
    		if(digit == 0|| n % digit != 0){
    			return false;
    		}
    		num /= 10;
    	}
    	return true;
    }
}
