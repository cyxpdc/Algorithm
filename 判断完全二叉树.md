public class IsCBT {
    public static class Node {
        public int value;
        public Node left;
        public Node right;

        public Node(int data) {
            this.value = data;
        }
    }

    public static boolean isCBT(Node head) {
        if (head == null) {
            return true;
        }
        Queue<Node> queue = new LinkedList<>();
        boolean leaf = false;
        Node l = null;
        Node r = null;
        queue.offer(head);
        while (!queue.isEmpty()) {
            head = queue.poll();
            l = head.left;
            r = head.right;
            if ((leaf && (l != null || r != null)) || //情况2
                    (l == null && r != null)) {//情况1
                return false;
            }
            //等效
            if (l != null) {
                queue.offer(l);
            }
            if (r != null) {
                queue.offer(r);
            }
            if(l == null || r == null){//不是左右两个孩子都全,代表后面的节点leaf都需要为true
                leaf = true;
            }
            /*if (l != null) {
                queue.offer(l);
            }
            if (r != null) {
                queue.offer(r);
            } else {//运行到这里，左孩子为空，右孩子不为空是不可能发生的，所以可以优化
                leaf = true;
            }*/
        }
        return true;
    }
}
