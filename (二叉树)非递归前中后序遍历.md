//非递归先序遍历(根-左-右)
public void prePrintTreeIterative(TreeNode<T> root){
	Stack<TreeNode<T>> stack = new Stack<>();
	if(root != null){
		//根节点入栈
		stack.push(root);
		while(!stack.empty()){
			root = stack.pop();
			System.out.println(root.value);
			//先入栈的后输出，所以右节点先入栈
			if(root.right != null){
				stack.push(root.right);
			}
			if(root.left != null){
				stack.push(root.left);
			}
		}
	}
}
  
 //非递归后序遍历
//对每一个节点，都是左-右-根
public void postPrintTreeIterative(TreeNode<T> root){
	Stack<TreeNode<T>> stack = new Stack<>();
	TreeNode<T> temp = root;
	while(root != null){
		for(;root.left != null;root = root.left){
			stack.push(root);
		}
		while(root.right == null || root.right == temp){
			System.out.println(root.value);
			temp = root;
			if(stack.empty()){
				return ;
			}
			root = stack.pop();
		}
		stack.push(root);
		root = root.right;
	}
}

//对每一个节点，都是左-根-右
public void inOrderPrintTreeIterative(TreeNode<T> root){
	Stack<TreeNode<T>> stack = new Stack<>();
	while(root != null){
		while(root != null){
			if(root.right != null){
				stack.push(root.right);
			}
			stack.push(root);
			root = root.left;
		}
		root = stack.pop();
		while(!stack.empty() && root.right == null){
			System.out.println(root.value);
			root = stack.pop();
		}
		System.out.println(root.value);
		if(!stack.empty()){
			root = stack.pop();
		}else{
			root = null;
		}
	}
}
